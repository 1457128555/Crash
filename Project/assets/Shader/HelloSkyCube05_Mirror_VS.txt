//  HelloSkyCube05_Mirror_VS

#include "CommonUniforms_VS.txt"

layout (location = 0) in vec3 vPos;
layout (location = 1) in vec3 vNor;

out vec4 oWorldPos;
out vec4 oWorldNor; 

void main()
{
    oWorldPos = uModel * vec4(vPos, 1.0);
    oWorldNor.xyz = mat3(transpose(inverse(uModel))) * vNor;

    gl_Position = uProjection * uView * oWorldPos;
}
